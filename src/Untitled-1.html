function validateForm() { const username = document.getElementById('username').value; const email = document.getElementById('email').value; const password = document.getElementById('password').value; if (username === '' || email === '' || password === '') { alert('All fields are required'); return false; } if (password.length < 6) { alert('Password must be at least 6 characters long'); return false; } // Simple email validation const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/; if (!emailPattern.test(email)) { alert('Please enter a valid email address'); return false; } return true; }
